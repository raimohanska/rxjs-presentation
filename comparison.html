DOM -> jQuery -> RxJs

DOM

document.getElementById("slide").onclick = function() { console.log("hello") }

jQuery

$("#slide").click(function() { console.log("hello") }) 
    var sum = 0
    $("#slide").click(function() { 
      sum += 1
      console.log("sum=" + sum) })

RxJs

$("#slide").toObservable("click").Subscribe(function() { console.log("hello")}) 
    $("#slide").toObservable("click")
        .Scan(0, function(acc, _) { return acc + 1} )
        .Subscribe(function(sum) { console.log("sum=" + sum)})
